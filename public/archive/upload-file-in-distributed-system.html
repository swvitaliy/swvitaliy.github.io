Загрузка файлов в распределенной системе

20 нояб. 2019 г.

В данной статье описан способ сохранения файла в распределенной системе. В качестве среды выполнения используется NodeJs. Так же, для хранения мета информации о загруженном файле используется MongoDb и mongoose для доступа к ней.

Задача состоит в том, чтобы принять файл от пользователя и каким-то образом обработать его.

L7 load balancer (nginx) получает запрос на загрузку файла и отправляет его на произвольный воркер. Этот воркер сохраняет файл в локальную файловую систему, сохраняет мета информацию о загруженном файле в БД и отдает ключик для этого файла (file_token), один для всей системы, само собой. Затем, пользователь берет этот ключик и отправляет в запрос на процессинг. Запрос от lb может прийти на произвольный воркер – может на тот же самый, а может и на другой. Задача состоит в том, чтобы получить этот файл на этом обработчике.

Для хранения мета информации о загруженном файле используется схема данных FileToken. Для чтения файла используется метод FileToken.readFile, который сначала смотрит, сохранен ли файл на локальной машине, и, если да, то вызывает метод fs.readFile. Если же это не так, то он смотрит на какой машине был сохранен файл, формирует из названия машины и файла url и запрашивает файл по HTTP (временная директория для загруженных файлов должна быть открыта, например, на nginx).

var fileTokenSchema = new Schema({
    fileName: String,
    hostname: String,
    token: String,
    tmpDir: String
});

fileTokenSchema.virtual('url').get(function() {
    return 'http://' + this.hostname + CONFIG.uploadFiles.baseDomain + '/_tmp/' + path.basename(this.fileName);
});

fileTokenSchema.virtual('localPath').get(function() {
    return (this.tmpDir || CONFIG.uploadFiles.tmpDir) + '/' + path.basename(this.fileName);
});

fileTokenSchema.methods.readFile = function() {
    if (this.hostname === (process.env.HOSTNAME || CONFIG.hostname)) {
        return Q.nfcall(fs.readFile, this.localPath);
    }
    return Q.nfcall(request, {method: 'GET', url: this.url, encoding: null}).spread(function(res, data) {
        if (res.statusCode !== 200) {
            throw new FileTokenError('Failed to load tmp file "'+this.url+'"');
        }
        return data;
    }.bind(this));
};

fileTokenSchema.pre('save', function(next) {
    this.hostname = process.env.HOSTNAME || CONFIG.hostname;
    next();

});

exports.FileToken = coreDB.model('FileToken', fileTokenSchema);

Для загрузки файла используется функция upload_temp_file:

module.exports.upload_temp_file = function (req, res) {
  if(req.files['filename']){
    var file = req.files['filename'];
    if(req.files['filename'].length>0) {
      file = req.files['filename'][0];
    }
    var tmpPath = file['path'];
    var mimeType = file['type'];
    if(allowedMimeTypes.indexOf(mimeType ) === -1){
      logger.error('wrong mime type %s', mimeType);
      const errorMessate = errorResponseBuilder(gbErrors.FILE_TYPE_NOT_VALID)
      res.send(errorMessate);
    }else{
      var now = new Date().getTime();
      var filetoken = new FileToken();
      var tmpBasename = path.basename(tmpPath);
      var uploadPromise = CONFIG.uploadFiles.localUpload ? Q() :
        Q.nfcall(fs.rename, tmpPath, path.join(CONFIG.uploadFiles.tmpDir, tmpBasename));
      uploadPromise.then(function() {
        filetoken.fileName = tmpPath;
        filetoken.token = crypto.createHmac('sha1', 'fileToken').update([tmpPath, now].join()).digest('hex');
        return filetoken.saveQ().then(function(){
          res.json({token:filetoken.token});
        });
      }).fail(function(err) {
        printErr(logger, err);
        const errorMessate = errorResponseBuilder(gbErrors.FILE_UPLOAD_FAIL)
        res.send(errorMessate);
      }).done();
    }
  }else{
    const errorMessate = errorResponseBuilder(gbErrors.FILE_NOT_FOUND)
    res.send(errorMessate);
  }
};

Для удаления мета информации о загруженном файле используется функция remove_temp_file:

module.exports.remove_temp_file = async function (req, res) {
  if (req.body.token) {
    const ft = await FileToken.findOneQ({ token: req.body.token });
    if (ft) {
      ft.removeQ();
    }
    res.json({ result: true });
  }
  res.json({result: true});
};

Удаление самих файлов из временной директории можно выполнять по крону, с использованием даты последней модификации файла.

Не используйте директорию /tmp для пусть даже и для временного хранения загруженных файлов, поскольку это может привести к потере данных.
